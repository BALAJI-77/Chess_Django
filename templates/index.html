import sneetch

# instantiate your machine learning model
model = YourAwesomeModel()

# connect to our API
env = sneetch.env(api_id=API_ID, api_key=API_KEY, task='turing_test')

# reinforcement learning loop
obs, done = env.init()
while not done:
    # choose the next action
    action = model.choose_action()

    # receive task feedback in OpenAI gym format
    obs, reward, done, info = env.step(action)

    # integrate feedback into the model
    model.observe(obs=obs, reward=reward, done=done)

# exam questions after learning
questions = env.exam_questions()
for question in questions:
    answer = model.answer_question(question)
    env.answer(question_id=question['id'], answer=answer)
